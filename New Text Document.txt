#include<bits/stdc++.h>
using namespace std;
 
int binary(int a[],int low,int high,int x)
{
while(low<=high)
    {int mid=(low+high)/2;
    if(a[mid]==x){return mid+1;
    }
    else if(x<a[mid])
    {
    return binary(a,low,mid-1,x);
    }
    else if(x>a[mid])
    {
    return binary(a,mid+1,high,x);
    }
    }
    return -1;
}
int main()
{
    int x,n;
cin>>n;
 
int a[n];
    
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    sort(a,a+n);
    cin>>x;
    while(x)
    {
        x--;
        int y;
        cin>>y;
       int xx= binary(a,0,n,y);
        cout<<xx<<endl;
        
    }
    
}
